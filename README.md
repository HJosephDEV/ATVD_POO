        product_price DOUBLE
    );
CREATE TABLE STOCK 
	(
        product_id INT,
        product_quantity INT,
        store_id INT
    );
ALTER TABLE STOCK
ADD FOREIGN KEY (product_id) REFERENCES Products(product_id),
ADD FOREIGN KEY (store_id) REFERENCES Stores(store_id);
